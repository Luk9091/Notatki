    \subsection{Transformacja Dyskretna (DTFT)}

                x_p = \sum\limits_{n=-\infty}^{+\infty}x(n\Delta t)\cdot \delta(t-n\Delta t) &&

            X(k) = \sum_{n=0}^{N-1}x(n)\cdot e^{-2j\pi \frac{k}{N}\cdot n} &&

            k =\ 0,\ 1,\ ...,\ N-1 && n =\ 0,\ 1,\ ...,\ N-1
        \end{align*}
