                x(t) \xleftrightarrow[\text{aaa}]{\text{bbb}} X(f)\\

            \begin{gather*}
                x(a t)\xleftrightarrow[\text{ICFT}]{\text{CFT}} \frac{1}{|a|} X(\frac{f}{a})\\
